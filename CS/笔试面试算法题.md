# 笔试面试算法题

### 五只球队的总分情况数

五支球队, 互相比赛, 胜者 + 3, 败者 + 0, 平局各 + 1 

问: 得分排序后共多少种情况

```C++
#include <iostream>
#include <cstring>
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>

#define x first
#define y second

using namespace std;

typedef pair<int, int> PII;

const int N = 5;

vector<PII> arr, socre = {{3, 0}, {0, 3}, {1, 1}};
int path[N + 1], tmp[N + 1];
unordered_set<string> res;

void dfs(int u)
{
    if(u == arr.size())
    {
        memcpy(tmp, path, sizeof path);
        sort(tmp, tmp + N + 1);
        
        string s;
        for(int i = 1; i <= N; i++)
            s += to_string(tmp[i]) + ',';
        res.insert(s);
        return;
    }
    
    for(int i = 0; i < socre.size(); i++)
    {
        int a = arr[u].x, b = arr[u].y;
        path[a] += socre[i].x, path[b] += socre[i].y;
        dfs(u + 1);
        path[a] -= socre[i].x, path[b] -= socre[i].y;
    }
}

int main()
{
    for(int i = 1; i <= N; i++)
        for(int j = i + 1; j <= N; j++)
            arr.push_back({i, j});
            
    dfs(0);
    
    cout << res.size() << endl;
    return 0;
}
```
